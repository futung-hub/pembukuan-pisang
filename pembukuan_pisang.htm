    <script>

        // Harga tetap

        const SELL_PRICE = 5000;

        const BUY_PRICE = 4000;

       

        // Data awal

        let data = {

            stock: 0,

            totalSold: 0,

            totalRevenue: 0,

            totalCapital: 0,

            transactions: [],

            dailySales: 0,

            lastReset: new Date().toDateString()

        };

       

        // Load data dari localStorage

        function loadData() {

            const saved = localStorage.getItem('bananaBookkeeping');

            if (saved) {

                data = JSON.parse(saved);

            }

            updateDisplay();

        }

       

        // Save data ke localStorage

        function saveData() {

            localStorage.setItem('bananaBookkeeping', JSON.stringify(data));

        }

       

        // Update tampilan

        function updateDisplay() {

            // Update dashboard

            document.getElementById('stock').textContent = data.stock;

            document.getElementById('totalSales').textContent = data.totalSold;

            document.getElementById('totalRevenue').textContent = formatCurrency(data.totalRevenue);

           

            const profit = data.totalRevenue - data.totalCapital;

            document.getElementById('profit').textContent = formatCurrency(profit);

            document.getElementById('profit').className = `card-value ${profit >= 0 ? 'profit' : 'loss'}`;

           

            // Update perhitungan real-time

            const buyQuantity = parseInt(document.getElementById('buyQuantity').value) || 0;

            document.getElementById('requiredCapital').textContent = formatCurrency(buyQuantity * BUY_PRICE);

           

            const sellQuantity = parseInt(document.getElementById('sellQuantity').value) || 0;

            document.getElementById('sellRevenue').textContent = formatCurrency(sellQuantity * SELL_PRICE);

           

            // Update transaction history

            updateTransactionHistory();

        }

       

        // Format currency

        function formatCurrency(amount) {

            return 'Rp' + amount.toLocaleString('id-ID');

        }

       

        // Beli stok

        function buyStock() {

            const quantity = parseInt(document.getElementById('buyQuantity').value);

           

            if (isNaN(quantity) || quantity <= 0) {

                alert('Masukkan jumlah yang valid!');

                return;

            }

           

            data.stock += quantity;

            data.totalCapital += quantity * BUY_PRICE;

           

            // Add transaction

            data.transactions.push({

                type: 'purchase',

                quantity: quantity,

                amount: quantity * BUY_PRICE,

                timestamp: new Date().toLocaleString()

            });

           

            // Keep only last 50 transactions

            if (data.transactions.length > 50) {

                data.transactions = data.transactions.slice(-50);

            }

           

            saveData();

            updateDisplay();

           

            alert(`✅ Berhasil membeli ${quantity} pisang\nModal: ${formatCurrency(quantity * BUY_PRICE)}`);

        }

       

        // Jual pisang

        function sellBananas() {

            const quantity = parseInt(document.getElementById('sellQuantity').value);

           

            if (isNaN(quantity) || quantity <= 0) {

                alert('Masukkan jumlah yang valid!');

                return;

            }

           

            if (quantity > data.stock) {

                alert(`❌ Stok tidak cukup!\nStok tersedia: ${data.stock} pisang`);

                return;

            }

           

            data.stock -= quantity;

            data.totalSold += quantity;

            data.totalRevenue += quantity * SELL_PRICE;

            data.dailySales += quantity;

           

            // Add transaction

            data.transactions.push({

                type: 'sale',

                quantity: quantity,

                amount: quantity * SELL_PRICE,

                timestamp: new Date().toLocaleString()

            });

           

            // Keep only last 50 transactions

            if (data.transactions.length > 50) {

                data.transactions = data.transactions.slice(-50);

            }

           

            saveData();

            updateDisplay();

           

            alert(`✅ Berhasil menjual ${quantity} pisang\nPendapatan: ${formatCurrency(quantity * SELL_PRICE)}`);

        }

       

        // Update transaction history

        function updateTransactionHistory() {

            const container = document.getElementById('transactionHistory');

            const transactions = data.transactions.slice(-10).reverse(); // Show last 10

           

            if (transactions.length === 0) {

                container.innerHTML = '<div class="transaction-item"><span>Belum ada transaksi</span><span></span></div>';

                return;

            }

           

            container.innerHTML = transactions.map(transaction => `

                <div class="transaction-item">

                    <div>

                        <span class="transaction-type ${transaction.type === 'sale' ? 'sale' : 'purchase'}">

                            ${transaction.type === 'sale' ? 'JUAL' : 'BELI'}

                        </span>

                        <span style="margin-left: 10px;">${transaction.quantity} pisang</span>

                    </div>

                    <div>

                        <span style="font-weight: bold;">${formatCurrency(transaction.amount)}</span>

                        <br>

                        <small style="color: #666;">${transaction.timestamp}</small>

                    </div>

                </div>

            `).join('');

        }

       

        // Rekomendasi pembelian

        function showRecommendation() {

            let recommendation = 0;

           

            if (data.dailySales > 0) {

                recommendation = Math.max(Math.ceil(data.dailySales * 1.2) - data.stock, 0);

            } else {

                recommendation = Math.max(20 - data.stock, 0);

            }

           

            if (recommendation > 0) {

                alert(`📊 REKOMENDASI PEMBELIAN\n\nDisarankan beli: ${recommendation} pisang\nPerkiraan modal: ${formatCurrency(recommendation * BUY_PRICE)}\n\nAlasan: Berdasarkan penjualan harian ${data.dailySales} pisang`);

            } else {

                alert(`📊 REKOMENDASI PEMBELIAN\n\nStok masih mencukupi (${data.stock} pisang)\nPenjualan hari ini: ${data.dailySales} pisang`);

            }

        }

       

        // Reset penjualan harian

        function resetDaily() {

            const today = new Date().toDateString();

           

            if (today !== data.lastReset) {

                data.dailySales = 0;

                data.lastReset = today;

                saveData();

                updateDisplay();

                alert('✅ Penjualan harian telah direset');

            } else {

                if (confirm('Reset penjualan hari ini? Data harian akan dikosongkan.')) {

                    data.dailySales = 0;

                    saveData();

                    updateDisplay();

                    alert('✅ Penjualan harian telah direset');

                }

            }

        }

       

        // Export data

        function exportData() {

            const exportData = {

                ...data,

                exportDate: new Date().toISOString(),

                priceInfo: {

                    sellPrice: SELL_PRICE,

                    buyPrice: BUY_PRICE

                }

            };

           

            const dataStr = JSON.stringify(exportData, null, 2);

            const dataUri = 'data:application/json;charset=utf-8,'+ encodeURIComponent(dataStr);

           

            const exportFileDefaultName = `pembukuan-pisang-${new Date().toISOString().split('T')[0]}.json`;

           

            const linkElement = document.createElement('a');

            linkElement.setAttribute('href', dataUri);

            linkElement.setAttribute('download', exportFileDefaultName);

            linkElement.click();

           

            alert('✅ Data berhasil diexport ke file JSON');

        }

       

        // Import data

        function importData() {

            const input = document.createElement('input');

            input.type = 'file';

            input.accept = '.json';

           

            input.onchange = e => {

                const file = e.target.files[0];

                const reader = new FileReader();

               

                reader.onload = event => {

                    try {

                        const imported = JSON.parse(event.target.result);

                       

                        // Validasi data

                        if (imported.stock !== undefined && imported.totalSold !== undefined) {

                            if (confirm('Ganti data saat ini dengan data yang diimport?')) {

                                data = imported;

                                saveData();

                                updateDisplay();

                                alert('✅ Data berhasil diimport');

                            }

                        } else {

                            alert('❌ File tidak valid!');

                        }

                    } catch (error) {

                        alert('❌ Error membaca file: ' + error.message);

                    }

                };

               

                reader.readAsText(file);

            };

           

            input.click();

        }

       

        // Event listeners untuk update real-time

        document.getElementById('buyQuantity').addEventListener('input', updateDisplay);

        document.getElementById('sellQuantity').addEventListener('input', updateDisplay);

       

        // Auto-check reset harian

        function checkDailyReset() {

            const today = new Date().toDateString();

            if (today !== data.lastReset) {

                data.dailySales = 0;

                data.lastReset = today;

                saveData();

                updateDisplay();

            }

        }

       

        // Initialize

        window.onload = function() {

            loadData();

            checkDailyReset();

           

            // Cek apakah hari sudah berganti setiap jam

            setInterval(checkDailyReset, 3600000); // 1 jam

        };

    </script>
